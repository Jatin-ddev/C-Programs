



// Prefix Operators (Increment and Dec )

#include<stdio.h>

void main(){

int num;    //10

printf("enter a number: ");
scanf("%d",&num);       

printf("%d\n",num); //10
printf("%d\n",++num); // 11
printf("%d\n",num); //11
printf("%d\n",--num); //10
printf("%d\n",num);   //10

}
